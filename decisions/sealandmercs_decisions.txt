# Switch between Cawa/Banner mercs:
country_decisions = {
	sealandmercs_switch_cawa_decision = {
		major = yes
		potential = {
			always = no # Turned off
			has_country_flag = sealandmercs_switch_merc_flag
			ai = no
		}
		allow = {
			is_at_war = no
			NOT = { num_of_banners = 1 }
		}
		effect = {
			clr_country_flag = sealandmercs_switch_merc_flag
			custom_tooltip = sealandmercs_switch_to_cawa_tt
			hidden_effect = {
				every_owned_province = {
					remove_province_modifier = sealand_banners_fl_base
					remove_province_modifier = sealand_cawa_fl_base
					clr_province_flag = sealand_banners_fl_base_flag
				}
				if = {
					limit = { NOT = { has_country_modifier = sealand_cawa_base_maintenance } }
					add_country_modifier = {
						name = sealand_cawa_base_maintenance
						duration = -1
						hidden = yes
					}
				}
				capital_scope = {
					remove_province_modifier = sealand_banners_fl_base
					add_province_modifier = {
						name = sealand_cawa_fl_base
						duration = -1
						hidden = yes
					}
				}
				#if = {
				#	limit = { is_capital = no }
				#	remove_province_modifier = sealand_banners_fl_base
				#	remove_province_modifier = sealand_cawa_fl_base
				#}
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
	sealandmercs_switch_banner_decision = {
		major = yes
		potential = {
			always = no # Turned off
			NOT = { has_country_flag = sealandmercs_switch_merc_flag }
			ai = no
		}
		allow = {
			is_at_war = no
			NOT = { num_of_cawa = 1 }
		}
		effect = {
			set_country_flag = sealandmercs_switch_merc_flag
			custom_tooltip = sealandmercs_switch_to_banner_tt
			hidden_effect = {
				every_owned_province = {
					remove_province_modifier = sealand_banners_fl_base
					remove_province_modifier = sealand_cawa_fl_base
				}
				#capital_scope = {
				#	remove_province_modifier = sealand_cawa_fl_base
				#}
				random_owned_province = {
					limit = {
						culture = ROOT
						is_core = ROOT
						is_state_core = root
					}
					set_province_flag = sealand_banners_fl_base_flag
					add_province_modifier = {
						name = sealand_banners_fl_base
						duration = -1
						hidden = yes
					}
				}
				#if = {
				#	limit = { is_capital = no }
				#	remove_province_modifier = sealand_banners_fl_base
				#	remove_province_modifier = sealand_cawa_fl_base
				#}
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
}
