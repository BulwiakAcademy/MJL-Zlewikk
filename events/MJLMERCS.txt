country_event = {
	id = mjl_merc_events.4
	title = "mjl_merc_events.EVTNAME4"
	desc = "mjl_merc_events.EVTDESC4"
	picture = COURT_eventPicture
	hidden = yes
	trigger = {
		ai = no
		any_owned_province = {
			OR = {
				has_province_flag = sealand_banners_fl_base_flag 
				has_province_modifier = sealand_banners_fl_base
			}
			OR = {
				NOT = { culture = ROOT }
				NOT = { is_state = yes }
				NOT = {	is_state_core = root }
			}
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
		every_owned_province = {
			limit = {
				OR = {
					has_province_flag = sealand_banners_fl_base_flag 
					has_province_modifier = sealand_banners_fl_base
				}
			}
			clr_province_flag = sealand_banners_fl_base_flag 
			remove_province_modifier = sealand_banners_fl_base
		}
		random_owned_province = {
			limit = {
				culture = ROOT
				is_core = ROOT
				is_state_core = ROOT
			}
			set_province_flag = sealand_banners_fl_base_flag 
			add_province_modifier = {
				name = sealand_banners_fl_base
				duration = -1
				hidden = yes
			}
		}
	}
	option = {}
}

country_event = {
	id = mjl_merc_events.10
	title = "mjl_merc_events.EVTNAME10"
	desc = "mjl_merc_events.EVTDESC10"
	picture = COURT_eventPicture
	
	trigger = {
		AND = {
			NOT = { has_country_flag = reestablish_the_immortals_flag }
			NOT = { tag = SAS }
			NOT = { tag = ACE }
		}
		AND = {
			NOT = { has_country_modifier = hun_black_army }
			NOT = { has_reform = black_army_reform }
		}
		NOT = { has_country_flag = mesopotamian_mamluks_flag }
		ai = no
		any_hired_mercenary_company = {
			hired_for_months = 1
		}
	}
	mean_time_to_happen = {
		days = 5
	}
	immediate = {
		hidden_effect = {
			random_hired_mercenary_company = {
				save_event_target_as = human_merc
			}
		}
	}
	option = {
		name = "mjl_merc_events.EVTOPTA10"
		event_target:human_merc = {
			disband_mercenary_company = THIS
		}
	}
}
