namespace = mjl_mongol_events
normal_or_historical_nations = yes

#Mongolia movespeed boost
country_event = {
	id = mjl_mongol_events.1
	title = none
	desc = none
	picture = none
	hidden = yes

	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		tag = KRC
	}

	immediate = {
		hidden_effect = {
			mongolia_region = {
				limit = {
					is_city = yes
					NOT = { area = cicigar_area }
					NOT = { area = eastern_mongolia }
					NOT = { area = chifeng_area }
				}
				add_province_modifier = {
					name = "mjl_mongolia_movespeed_boost"
					hidden = yes
					duration = -1
				}
			}
		}
	}
	option = { }
}

#Esen Death
country_event = {
	id = mjl_mongol_events.2
	title = "mjl_mongol_events.2.t"
	desc = "mjl_mongol_events.2.d"
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes

	trigger = {
		tag = ZUN
		NOT = { has_global_flag = mjl_esen_choros_death_global_flag }
	}

	immediate = {
		hidden_effect = {
			set_global_flag = mjl_esen_choros_death_global_flag
		}
	}

	option = { 
		name = "mjl_mongol_events.3.a" # My time has come!
		add_prestige = 10
		add_country_modifier = {
			name = "mjl_esen_choros_death_modifier"
			duration = 1825
		}
		OIR = {
			every_subject_country = {
				add_liberty_desire = 100
			}
			if = {
				limit = {
					SYG = {
						is_subject_of = OIR
					}
				}
				free_vassal = SYG
			}
			if = {
				limit = {
					TUM = {
						is_subject_of = OIR
					}
				}
				free_vassal = TUM
			}
			if = {
				limit = {
					HMI = {
						is_subject_of = OIR
					}
				}
				free_vassal = HMI
			}
		}
	}
}
